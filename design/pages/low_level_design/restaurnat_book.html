Restaurant Management System

This design outlines a Restaurant Management System that allows customers to interact with the restaurant and enables the restaurant to manage orders, inventory, staff, and analytics efficiently.

Key Features and Requirements:

  • Customers can:
    • View the menu.
    • Place orders.
    • Make reservations.
  • The restaurant can:
    • Manage inventory (ingredients and menu items).
    • Process orders, including preparation, billing, and payment.
    • Support multiple payment methods (cash, credit card, mobile payments).
    • Manage staff details, schedules, and performance tracking.
    • Generate reports and analytics.
  • The system should:
    • Handle concurrent access and ensure data consistency.
    • Provide APIs or interfaces for smooth interaction between components.

Class Design:

The following are the main components of the system, along with their functionality:

1. MenuItem Class

Represents an item in the menu.


class MenuItem {
    private String name;
    private double price;
    private String category;

    public MenuItem(String name, double price, String category) {
        this.name = name;
        this.price = price;
        this.category = category;
    }

    public String getName() {
        return name;
    }

    public double getPrice() {
        return price;
    }

    public String getCategory() {
        return category;
    }
}

3. Order Class

Represents an order placed by a customer.


import java.util.List;

class Order {
    private int orderId;
    private Customer customer;
    private List items;
    private double totalAmount;
    private String paymentMethod;

    public Order(int orderId, Customer customer, List items, String paymentMethod) {
        this.orderId = orderId;
        this.customer = customer;
        this.items = items;
        this.paymentMethod = paymentMethod;
        this.totalAmount = calculateTotal();
    }

    private double calculateTotal() {
        return items.stream().mapToDouble(MenuItem::getPrice).sum();
    }

    public void displayOrderSummary() {
        System.out.println("Order ID: " + orderId);
        System.out.println("Customer: " + customer.getName());
        System.out.println("Items:");
        for (MenuItem item : items) {
            System.out.println("- " + item.getName() + ": $" + item.getPrice());
        }
        System.out.println("Total Amount: $" + totalAmount);
        System.out.println("Payment Method: " + paymentMethod);
    }
}
4. Inventory Class

Tracks ingredients and their quantities.


import java.util.HashMap;
import java.util.Map;

class Inventory {
    private Map ingredients = new HashMap<>();

    public void addIngredient(String ingredient, int quantity) {
        ingredients.put(ingredient, ingredients.getOrDefault(ingredient, 0) + quantity);
    }

    public boolean useIngredient(String ingredient, int quantity) {
        if (ingredients.containsKey(ingredient) && ingredients.get(ingredient) >= quantity) {
            ingredients.put(ingredient, ingredients.get(ingredient) - quantity);
            return true;
        }
        return false;
    }

    public void displayInventory() {
        System.out.println("Current Inventory:");
        for (Map.Entry entry : ingredients.entrySet()) {
            System.out.println("- " + entry.getKey() + ": " + entry.getValue());
        }
    }
}
5. Reservation Class

Handles table reservations for customers.


import java.util.HashMap;
import java.util.Map;

class Reservation {
    private Map reservations = new HashMap<>();

    public boolean reserveTable(String customerName, String timeSlot) {
        if (!reservations.containsKey(timeSlot)) {
            reservations.put(timeSlot, customerName);
            return true;
        }
        return false;
    }

    public void displayReservations() {
        System.out.println("Reservations:");
        for (Map.Entry entry : reservations.entrySet()) {
            System.out.println("- " + entry.getKey() + ": Reserved by " + entry.getValue());
        }
    }
}
Sample Input and Output:

The following demonstrates a simple flow:


public class RestaurantApp {
    public static void main(String[] args) {
        // Inventory setup
        Inventory inventory = new Inventory();
        inventory.addIngredient("Cheese", 10);
        inventory.addIngredient("Tomato", 15);

        // Menu setup
        MenuItem pizza = new MenuItem("Pizza", 12.5, "Main Course");
        MenuItem salad = new MenuItem("Salad", 5.0, "Appetizer");

        // Customer and Order
        Customer customer = new Customer("Alice", "123-456-7890");
        Order order = new Order(1, customer, List.of(pizza, salad), "Credit Card");

        // Display Order Summary
        order.displayOrderSummary();

        // Reservation
        Reservation reservation = new Reservation();
        if (reservation.reserveTable("Alice", "7:00 PM")) {
            System.out.println("Reservation successful for 7:00 PM.");
        } else {
            System.out.println("Time slot already reserved.");
        }

        // Display Reservations
        reservation.displayReservations();

        // Inventory after usage
        inventory.useIngredient("Cheese", 2);
        inventory.useIngredient("Tomato", 3);
        inventory.displayInventory();
    }
}
